385 * must do a Skip_Blanks before we exit, so that the
386 * next call (or the code that picks up the cmd) can
387 * assume the same thing.
388 */
389
390 Debug(DPARS|DEXT, ("get_list()...entered\n"));
391
392 /* list = range {"," range}
393 */
394
395 /* clear the bit string, since the default is 'off'.
396 */
397 bit_nclear(bits, 0, (high-low+1)){ register bitstr_t *_name = bits; register int _start = 0, _stop
= (high-low+1); register int _startbyte = ((_start) >>
3); register int _stopbyte = ((_stop) >> 3); if (_startbyte
== _stopbyte) { _name[_startbyte] &= ((0xff >> (8 -
(_start&0x7))) | (0xff << ((_stop&0x7) + 1)));
} else { _name[_startbyte] &= 0xff >> (8 - (_start
&0x7)); while (++_startbyte < _stopbyte) _name[_startbyte
] = 0; _name[_stopbyte] &= 0xff << ((_stop&0x7)
+ 1); } };
398
399 /* process all ranges
400 */
401 done = FALSE0;
402 while (!done) {
403 ch = get_range(bits, low, high, names, ch, file);
404 if (ch == ',')
405 ch = get_char(file);
406 else
407 done = TRUE1;
408 }
409
410 /* exiting. skip to some blanks, then skip over the blanks.
411 */
412 Skip_Nonblanks(ch, file)while (ch!='\t' && ch!=' ' && ch!='\n' &&
ch != (-1)) ch = get_char(file);
413 Skip_Blanks(ch, file)while (ch == '\t' || ch == ' ') ch = get_char(file);
414
415 Debug(DPARS|DEXT, ("get_list()...exiting w/ %02x\n", ch));
416
417 return ch;
418}

421static char
422get_range(bits, low, high, names, ch, file)
423 bitstr_t *bits; /* one bit per flag, default=FALSE */
424 int low, high; /* bounds, impl. offset for bitstr */
425 char *names[]; /* NULL or names of elements */
426 int ch; /* current character being processed */
427 FILE *file; /* file being read */
428{
429 /* range = number | number "-" number [ "/" number ]
430 */
431
432 register int i;
433 auto int num1, num2, num3;
434
435 Debug(DPARS|DEXT, ("get_range()...entering, exit won't show\n"));
436
437 if (ch == '*') {
438 /* '*' means "first-last" but can still be modified by /step
439 */
440 num1 = low;
441 num2 = high;
442 ch = get_char(file);
443 if (ch == EOF(-1))
444 return EOF(-1);
445 } else {
446 if (EOF(-1) == (ch = get_number(&num1, low, names, ch, file)))
447 return EOF(-1);
448
449 if (ch != '-') {
450 /* not a range, it's a single number.
451 */
452
453 /* Unsupported syntax: Step specified without range,
454 eg: 1/20 * * * * /bin/echo "this fails"
455 */
456 if (ch == '/')
457 return EOF(-1);
458
459 if (EOF(-1) == set_element(bits, low, high, num1))
460 return EOF(-1);
461 return ch;
462 } else {
463 /* eat the dash
464 */
465 ch = get_char(file);
466 if (ch == EOF(-1))
467 return EOF(-1);
468
469 /* get the number following the dash
470 */
471 ch = get_number(&num2, low, names, ch, file);
472 if (ch == EOF(-1))
473 return EOF(-1);
474 }
475 }
476
477 /* check for step size
478 */
479 if (ch == '/') {
480 /* eat the slash
481 */
482 ch = get_char(file);
483 if (ch == EOF(-1))
484 return EOF(-1);
485
486 /* get the step size -- note: we don't pass the
487 * names here, because the number is not an
488 * element id, it's a step size. 'low' is
489 * sent as a 0 since there is no offset either.
490 */
491 ch = get_number(&num3, 0, PPC_NULL((char **)((void*)0)), ch, file);
492 if (ch == EOF(-1) || num3 <= 0)
493 return EOF(-1);
494 } else {
495 /* no step. default==1.
496 */
497 num3 = 1;
498 }
499
500 /* Explicitly check for sane values. Certain combinations of ranges and
501 * steps which should return EOF don't get picked up by the code below,
502 * eg:
503 * 5-64/30 * * * * touch /dev/null
504 *
505 * Code adapted from set_elements() where this error was probably intended
506 * to be catched.
507 */
508 if (num1 < low || num1 > high || num2 < low || num2 > high)
509 return EOF(-1);
510
511 /* range. set all elements from num1 to num2, stepping
512 * by num3. (the step is a downward-compatible extension
513 * proposed conceptually by bob@acornrc, syntactically
514 * designed then implmented by paul vixie).
515 */
516 for (i = num1; i <= num2; i += num3)
517 if (EOF(-1) == set_element(bits, low, high, i))
518 return EOF(-1);
519
520 return ch;
521}

524static char
525get_number(numptr, low, names, ch, file)
526 int *numptr; /* where does the result go? */
527 int low; /* offset applied to result if symbolic enum used */
528 char *names[]; /* symbolic names, if any, for enums */
529 int ch; /* current character */
530 FILE *file; /* source */
531{
532 char temp[MAX_TEMPSTR1000], *pc;
533 int len, i, all_digits;
534
535 /* collect alphanumerics into our fixed-size temp array
536 */
537 pc = temp;
538 len = 0;
539 all_digits = TRUE1;
540 while (isalnum(ch)((*__ctype_b_loc ())[(int) ((ch))] & (unsigned short int)
_ISalnum)) {
541 if (++len >= MAX_TEMPSTR1000)
542 return EOF(-1);
543
544 *pc++ = ch;
545
546 if (!isdigit(ch)((*__ctype_b_loc ())[(int) ((ch))] & (unsigned short int)
_ISdigit))
547 all_digits = FALSE0;
548
549 ch = get_char(file);
550 }
551 *pc = '\0';
552
553 if (len == 0) {
554 return EOF(-1);
555 }
556
557 /* try to find the name in the name list
558 */
559 if (names) {
560 for (i = 0; names[i] != NULL((void*)0); i++) {
561 Debug(DPARS|DEXT,;
562 ("get_num, compare(%s,%s)\n", names[i], temp));
563 if (!strcasecmp(names[i], temp)) {
564 *numptr = i+low;
565 return ch;
566 }
567 }
568 }
569
570 /* no name list specified, or there is one and our string isn't
571 * in it. either way: if it's all digits, use its magnitude.
572 * otherwise, it's an error.
573 */
574 if (all_digits) {
575 *numptr = atoi(temp);
576 return ch;
577 }
578
579 return EOF(-1);
580}
